function rechercher() { let id = document.getElementById("id").value.trim(); if (!id) { document.getElementById("result").innerHTML = "⚠️ Veuillez entrer un ID"; return; } document.getElementById("result").innerHTML = "⏳ Recherche en cours..."; fetch("https://opensheet.elk.sh/1IHlqsby69hHzrbp5Ed-5BCH3AbDi4mliD4kmB7BB5bA/API") .then(res => res.json()) .then(data => { let resultats = data.filter(row => (row["Submission ID"] || "").trim() === id ); if (resultats.length === 0) { document.getElementById("result").innerHTML = "❌ Aucun dossier trouvé"; return; } let html = ""; resultats.forEach(d => { html += `
ID : ${d["Submission ID"]}
Date de soumission : ${d["Submitted at"] || ""}
Prénom : ${d["Prénom"] || ""}
Nom : ${d["Nom"] || ""}
Date de naissance : ${d["Date de naissance"] || ""}
Nationalité : ${d["Nationalité"] || ""}
Poste : ${d["Poste souhaité"] || ""}
Statut : ${d["Statut"] || ""}
Commentaire : ${d["Commentaire"] || ""}
`; }); document.getElementById("result").innerHTML = "✅ Dossier trouvé
" + html; }) .catch(error => { document.getElementById("result").innerHTML = "❌ Erreur de connexion"; console.error(error); }); }
