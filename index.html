function rechercher() { let id = document.getElementById("id").value.trim(); if (!id) { document.getElementById("result").innerHTML = "⚠️ Veuillez entrer un ID"; return; } document.getElementById("result").innerHTML = "⏳ Recherche en cours..."; fetch("https://opensheet.elk.sh/1IHlqsby69hHzrbp5Ed-5BCH3AbDi4mliD4kmB7BB5bA/API") .then(res => res.json()) .then(data => { let resultats = data.filter(row => (row["Submission ID"] || "").toLowerCase().trim() === id.toLowerCase() ); if (resultats.length === 0) { document.getElementById("result").innerHTML = "❌ Aucun dossier trouvé"; return; } let html = ""; resultats.forEach(d => { // 🎯 Statut let statut = (d["Statut"] || "").toLowerCase(); let statutCouleur = "gray"; let statutTexte = d["Statut"] || "En attente"; if (statut === "valide") { statutCouleur = "green"; } else if (statut === "en cours") { statutCouleur = "orange"; } // 📅 Date sécurisée (IMPORTANT) let dateSoumission = "Non disponible"; if (d["Submitted at"]) { let dateObj = new Date(d["Submitted at"]); if (!isNaN(dateObj)) { dateSoumission = dateObj.toLocaleString("fr-FR"); } } html += `

📄 Dossier candidat

ID : ${d["Submission ID"] || "-"}

Date de soumission : ${dateSoumission}

Prénom : ${d["Prénom"] || "-"}

Nom : ${d["Nom"] || "-"}

Date de naissance : ${d["Date de naissance"] || "-"}

Nationalité : ${d["Nationalité"] || "-"}

Poste : ${d["Poste souhaité"] || "-"}

Statut : ${statutTexte}

Commentaire : ${d["Commentaire"] || "Aucun"}

`; }); document.getElementById("result").innerHTML = `
🔍 Résultat pour ID : ${id}
${html} `; }) .catch(error => { document.getElementById("result").innerHTML = "❌ Erreur de connexion"; console.error(error); }); }
